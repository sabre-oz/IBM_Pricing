<script>

document.addEventListener('DOMContentLoaded', () => {

    // --- DATA & CONFIG ---

    const PROFILE_PRICING = {

            { Profile: 'XS 4x16', vCPUs: 4, RAM_GB: 16, Cost_Monthly_USD: 630.36 }, { Profile: 'S 8x32', vCPUs: 8, RAM_GB: 32, Cost_Monthly_USD: 1260.72 }, { Profile: 'S+ 8x64', vCPUs: 8, RAM_GB: 64, Cost_Monthly_USD: 1656.24 }, { Profile: 'M 16x64', vCPUs: 16, RAM_GB: 64, Cost_Monthly_USD: 2521.44 }, { Profile: 'L 32x128', vCPUs: 32, RAM_GB: 128, Cost_Monthly_USD: 5042.88 }, { Profile: 'XL 30x240', vCPUs: 30, RAM_GB: 240, Cost_Monthly_USD: 6210.90 },

        ],

    };

    const STORAGE_COST_PER_GB_MONTH = 0.63;

    const SUBSCRIPTION_DURATIONS = [

        { id: 1, name: '1 Month' }, { id: 12, name: '1 Year (12 Months)' }, { id: 36, name: '3 Years (36 Months)' },

    ];

    const DATABASE_OPTIONS = Object.keys(PROFILE_PRICING);


    // --- STATE ---

    let state = {

        db: DATABASE_OPTIONS[0],

        profile: PROFILE_PRICING[DATABASE_OPTIONS[0]][0].Profile,

        storage: 256,

        duration: 12,

    };

    let comparisons = [];

    let nextId = 1;

    let barChartInstance = null;

    let lineChartInstance = null;


    // --- DOM ELEMENTS ---

    const dbSelect = document.getElementById('db-select');

    const profileSelect = document.getElementById('profile-select');

    const storageSlider = document.getElementById('storage-slider');

    const storageValue = document.getElementById('storage-value');

    const durationSelect = document.getElementById('duration-select');

    const totalCostDisplay = document.getElementById('total-cost-display');

    const monthlyCostDisplay = document.getElementById('monthly-cost-display');

    const addToComparisonBtn = document.getElementById('add-to-comparison-btn');

    const comparisonTableBody = document.getElementById('comparison-table-body');

    const placeholder = document.getElementById('placeholder');

    const comparisonContent = document.getElementById('comparison-content');

    

    // --- HELPER FUNCTIONS ---

    const formatCurrency = (amount) => new Intl.NumberFormat('en-US', { style: 'currency', currency: 'USD' }).format(amount);


    // --- CORE LOGIC ---

    function calculateCost(config) {

        const profileData = PROFILE_PRICING[config.db]?.find(p => p.Profile === config.profile);

        if (!profileData) return { monthly: 0, total: 0 };

        const profileMonthlyCost = profileData.Cost_Monthly_USD;

        const storageMonthlyCost = config.storage * STORAGE_COST_PER_GB_MONTH;

        const monthlyTotal = profileMonthlyCost + storageMonthlyCost;

        const total = monthlyTotal * config.duration;

        return { monthly: monthlyTotal, total: total };

    }

    

    // --- UI UPDATE FUNCTIONS ---

    function updateConfigCost() {

        const cost = calculateCost(state);

        totalCostDisplay.textContent = formatCurrency(cost.total);

        monthlyCostDisplay.textContent = `${formatCurrency(cost.monthly)} / month`;

    }


    function populateDbSelect() {

        dbSelect.innerHTML = DATABASE_OPTIONS.map(db => `<option value="${db}">${db}</option>`).join('');

    }


    function populateDurationSelect() {

        durationSelect.innerHTML = SUBSCRIPTION_DURATIONS.map(d => `<option value="${d.id}">${d.name}</option>`).join('');

        durationSelect.value = state.duration;

    }


    function populateProfileSelect() {

        const profiles = PROFILE_PRICING[state.db];

        profileSelect.innerHTML = profiles.map(p => `<option value="${p.Profile}">${p.Profile} (${p.vCPUs} vCPU, ${p.RAM_GB}GB RAM)</option>`).join('');

        // Ensure state profile is valid for the selected DB

        if (!profiles.some(p => p.Profile === state.profile)) {

            state.profile = profiles[0].Profile;

        }

        profileSelect.value = state.profile;

    }

    

    function renderComparison() {

        // Toggle placeholder

        if (comparisons.length > 0) {

            placeholder.classList.add('hidden');

            comparisonContent.classList.remove('hidden');

        } else {

            placeholder.classList.remove('hidden');

            comparisonContent.classList.add('hidden');

            return;

        }


        // Render table

        comparisonTableBody.innerHTML = comparisons.map(item => `

            <tr class="hover:bg-slate-50">

                <td class="px-4 py-4 whitespace-nowrap text-sm font-semibold text-slate-900">${item.db}</td>

                <td class="px-4 py-4 whitespace-nowrap text-sm font-medium text-slate-800">${formatCurrency(item.cost.monthly)}</td>

                </td>

            </tr>

        `).join('');


        // Render charts

        renderBarChart();

        renderLineChart();

    }

    

    function renderBarChart() {

        const ctx = document.getElementById('barChart').getContext('2d');

        const data = {

                    data: comparisons.map(c => c.cost.total),

                    borderColor: 'rgba(79, 70, 229, 1)',

                    borderWidth: 1,

                    borderRadius: 4,

                },

                {

                    label: 'Monthly Cost',

                    data: comparisons.map(c => c.cost.monthly),

                    backgroundColor: 'rgba(129, 140, 248, 0.8)',

                    borderColor: 'rgba(129, 140, 248, 1)',

                }

            }

        });

    }


    function renderLineChart() {

        const ctx = document.getElementById('lineChart').getContext('2d');

        const lineColors = ['#4f46e5', '#10b981', '#f59e0b', '#ef4444', '#3b82f6', '#8b5cf6'];

        const maxDuration = comparisons.length > 0 ? Math.max(...comparisons.map(c => c.duration)) : 0;

        const labels = Array.from({ length: maxDuration }, (_, i) => `Month ${i + 1}`);

        const datasets = comparisons.map((item, index) => {

            const data = [];

            for (let month = 1; month <= maxDuration; month++) {

                data.push(month <= item.duration ? item.cost.monthly * month : null);

            }

            return {

                label: `${item.db.split(' ')[0]} #${item.id}`,

                data: data,

                borderColor: lineColors[index % lineColors.length],

                backgroundColor: lineColors[index % lineColors.length] + '33', // a transparent version

                fill: false,

                tension: 0.1,

                spanGaps: true

            };

        });


        if (lineChartInstance) {

            lineChartInstance.destroy();

        }

        lineChartInstance = new Chart(ctx, {

            type: 'line',

            data: { labels, datasets },

            options: {

                responsive: true,

                maintainAspectRatio: false,

                scales: {

                    y: { 

                        beginAtZero: true, 

                        ticks: { callback: value => formatCurrency(value) } 

                    }

                },

                plugins: {

                    tooltip: {

                        callbacks: { label: context => `${context.dataset.label}: ${formatCurrency(context.raw)}` }

                    }

                }

            }

        });

    }

        updateConfigCost();

    });

    storageSlider.addEventListener('input', (e) => {

        state.storage = Number(e.target.value);

        storageValue.textContent = `${state.storage} GB`;

        updateConfigCost();

    });

    durationSelect.addEventListener('change', (e) => {

        state.duration = Number(e.target.value);

        updateConfigCost();

    });

    addToComparisonBtn.addEventListener('click', () => {

        const profileData = PROFILE_PRICING[state.db].find(p => p.Profile === state.profile);

        const newComparison = {

            id: nextId++,

            ...state,

            vcpu: profileData.vCPUs,

            ram: profileData.RAM_GB,

            cost: calculateCost(state),

        };

        comparisons.push(newComparison);

        renderComparison();

    });

    comparisonTableBody.addEventListener('click', (e) => {

        if (e.target.classList.contains('remove-btn')) {

            const idToRemove = Number(e.target.dataset.id);

            comparisons = comparisons.filter(item => item.id !== idToRemove);

            renderComparison();

        }

    });


    // --- INITIALIZATION ---

    function init() {

        populateDbSelect();

        populateProfileSelect();

        populateDurationSelect();

        updateConfigCost();

    }


    init();

});
